CREATE TABLE BOARD(
                      NO int not null auto_increment primary key,
                      TITLE varchar(30) not null,
                      CATEGORY varchar(20) not null,
                      WRITER varchar(10) not null,
                      PASSWORD varchar(20) not null,
                      CONTENTS varchar(1000),
                      LIKECNT int(10),
                      HATECNT int(10),
                      HITCNT int(10),
                      FILENAME varchar(50),
                      regdate DATE);

select * from BOARD;
